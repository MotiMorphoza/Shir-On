backend/src/utils/normalize.js

`js
/*
  Normalize a string for matching:
  lowercase, strip punctuation, collapse whitespace,
  remove "feat.", "ft.", "featuring" credits.
 /
export function normalize(str = '') {
  return str
    .toLowerCase()
    .replace(/\(feat\..?\)/gi, '')
    .replace(/\[feat\..?\]/gi, '')
    .replace(/feat\..?(?=\s|$)/gi, '')
    .replace(/\bft\..?(?=\s|$)/gi, '')
    .replace(/\bfeaturing\b.?(?=\s|$)/gi, '')
    .replace(/[^\w\s]/g, '')
    .replace(/\s+/g, ' ')
    .trim();
}

/*
  Similarity ratio between two normalized strings (0–1).
  Simple Levenshtein-based ratio for confidence scoring.
 /
export function similarity(a, b) {
  const na = normalize(a);
  const nb = normalize(b);
  if (na === nb) return 1;
  const dist = levenshtein(na, nb);
  const maxLen = Math.max(na.length, nb.length);
  return maxLen === 0 ? 1 : 1 - dist / maxLen;
}

function levenshtein(a, b) {
  const m = a.length, n = b.length;
  const dp = Array.from({ length: m + 1 }, (, i) =>
    Array.from({ length: n + 1 }, (, j) => (i === 0 ? j : j === 0 ? i : 0))
  );
  for (let i = 1; i <= m; i++)
    for (let j = 1; j <= n; j++)
      dp[i][j] = a[i-1] === b[j-1]
        ? dp[i-1][j-1]
        : 1 + Math.min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]);
  return dp[m][n];
}
`

backend/src/providers/spotify.js

`js
import axios from 'axios';

const BASE = 'https://api.spotify.com/v1';

export class SpotifyProvider {
  #token = null;
  #tokenExpiry = 0;

  constructor({ clientId, clientSecret }) {
    this.clientId = clientId;
    this.clientSecret = clientSecret;
  }

  async #getToken() {
    if (this.#token && Date.now() < this.#tokenExpiry) return this.#token;

    const res = await axios.post(
      'https://accounts.spotify.com/api/token',
      new URLSearchParams({ granttype: 'clientcredentials' }),
      {
        headers: {
          'Content-Type': 'application/x-www-form-urlencoded',
          Authorization: Basic ${Buffer.from(
            ${this.clientId}:${this.clientSecret}
          ).toString('base64')},
        },
      }
    );

    this.#token = res.data.accesstoken;
    this.#tokenExpiry = Date.now() + res.data.expiresin  1000 - 5000;
    return this.#token;
  }

  async #get(path, params = {}) {
    const token = await this.#getToken();
    const res = await axios.get(${BASE}${path}, {
      headers: { Authorization: Bearer ${token} },
      params,
    });
    return res.data;
  }

  /
    Fetch all tracks from a playlist (handles pagination).
   /
  async getPlaylistTracks(playlistId) {
    const tracks = [];
    let url = /playlists/${playlistId}/tracks;
    let params = { limit: 100, offset: 0 };

    while (url) {
      const data = await this.#get(url.replace(BASE, ''), params);
      for (const item of data.items) {
        if (item.track && item.track.id) tracks.push(this.#mapTrack(item.track));
      }
      // Spotify returns full next URL; we only need path+params handled above
      url = data.next ? null : null; // pagination via offset
      if (data.next) {
        params = { ...params, offset: params.offset + 100 };
        url = /playlists/${playlistId}/tracks;
      } else {
        url = null;
      }
    }

    return tracks;
  }

  async getAlbumTracks(albumId) {
    const album = await this.#get(/albums/${albumId});
    return album.tracks.items.map(t => this.#mapTrack(t, album));
  }

  async getPlaylistMeta(playlistId) {
    return this.#get(/playlists/${playlistId}, {
      fields: 'id,name,description,images',
    });
  }

  #mapTrack(track, albumOverride = null) {
    const album = albumOverride || track.album || {};
    return {
      spotifyid:   track.id,
      spotifyurl:  track.externalurls?.spotify || null,
      title:        track.name,
      artist:       track.artists?.map(a => a.name).join(', ') || '',
      album:        album.name  || '',
      albumspotifyid: album.id || null,
      tracknumber: track.tracknumber || null,
      year:         parseInt((album.releasedate || '0').slice(0, 4)) || null,
      coverurl:    album.images?.[0]?.url || null,
    };
  }
}

backend/src/providers/lyrics/lyricsovh.js

`js
import axios from 'axios';
import { BaseLyricsProvider } from './base.js';
import { normalize, similarity } from '../../utils/normalize.js';

/
  Provider 1: lyrics.ovh (free, no key required)
  Docs: https://lyricsovh.docs.apiary.io/
 /
export class LyricsOvhProvider extends BaseLyricsProvider {
  get name() { return 'lyrics.ovh'; }

  async fetch(title, artist) {
    try {
      const res = await axios.get(
        https://api.lyrics.ovh/v1/${encodeURIComponent(artist)}/${encodeURIComponent(title)},
        { timeout: 8000 }
      );

      const text = res.data?.lyrics?.trim();
      if (!text || text.length < 20) return null;

      // Rough confidence: how well do the returned path parts match our query
      const score = (similarity(title, title) + similarity(artist, artist)) / 2;

      return {
        lyricstext:      text,
        source:           this.name,
        confidencescore: score,
      };
    } catch {
      return null;
    }
  }
}

backend/src/providers/lyrics/lrclib.js

`js
import axios from 'axios';
import { BaseLyricsProvider } from './base.js';
import { normalize, similarity } from '../../utils/normalize.js';

/*
  Provider 2: lrclib.net — free, open lyric database
  Docs: https://lrclib.net/docs
 /
export class LrclibProvider extends BaseLyricsProvider {
  get name() { return 'lrclib.net'; }

  async fetch(title, artist) {
    try {
      const res = await axios.get('https://lrclib.net/api/search', {
        params: { trackname: title, artistname: artist },
        timeout: 8000,
      });

      const results = res.data;
      if (!Array.isArray(results) || results.length === 0) return null;

      // Pick the best match by title+artist similarity
      const scored = results
        .filter(r => r.plainLyrics)
        .map(r => ({
          ...r,
          score:
            (similarity(title, r.trackName) + similarity(artist, r.artistName)) / 2,
        }))
        .sort((a, b) => b.score - a.score);

      if (!scored.length || scored[0].score < 0.4) return null;

      return {
        lyricstext:      scored[0].plainLyrics.trim(),
        source:           this.name,
        confidencescore: scored[0].score,
      };
    } catch {
      return null;
    }
  }
}
`

backend/src/providers/lyrics/index.js

`js
import { LyricsOvhProvider } from './lyricsovh.js';
import { LrclibProvider }    from './lrclib.js';

const PROVIDERS = [
  new LrclibProvider(),
  new LyricsOvhProvider(),
];

const CONFIDENCETHRESHOLD = 0.5;

/*
  Try each provider in order.
  Returns the first result above the confidence threshold,
  or the best result found, or null if all fail.
 /
export async function fetchLyrics(title, artist) {
  let best = null;

  for (const provider of PROVIDERS) {
    try {
      const result = await provider.fetch(title, artist);
      if (!result) continue;

      if (result.confidencescore >= CONFIDENCETHRESHOLD) return result;
      if (!best || result.confidencescore > best.confidencescore) best = result;
    } catch {
      // provider failed silently — try next
    }
  }

  return best; // may be null
}
`

backend/src/services/songService.js

`js
import { v4 as uuidv4 } from 'uuid';
import db               from '../db/index.js';
import { normalize }    from '../utils/normalize.js';
import { sanitizeText } from '../utils/sanitize.js';
import { fetchLyrics }  from '../providers/lyrics/index.js';

const CONFIDENCEREVIEWTHRESHOLD = 0.6;

// ─── Helpers ────────────────────────────────────────────────────────────────

function upsertArtist(name) {
  const norm = normalize(name);
  const existing = db.prepare(
    'SELECT  FROM artists WHERE normalized = ?'
  ).get(norm);
  if (existing) return existing;

  const id = uuidv4();
  db.prepare(
    'INSERT INTO artists (id, name, normalized) VALUES (?, ?, ?)'
  ).run(id, sanitizeText(name), norm);
  return db.prepare('SELECT  FROM artists WHERE id = ?').get(id);
}

function upsertAlbum({ title, artistId, year, coverUrl, spotifyId }) {
  const norm = normalize(title);
  const existing = db.prepare(
    'SELECT  FROM albums WHERE normalized = ? AND artistid = ?'
  ).get(norm, artistId);
  if (existing) return existing;

  const id = uuidv4();
  db.prepare(
    INSERT INTO albums (id, title, normalized, artistid, year, coverurl, spotifyid)
     VALUES (?, ?, ?, ?, ?, ?, ?)
  ).run(id, sanitizeText(title), norm, artistId, year, coverUrl, spotifyId);
  return db.prepare('SELECT  FROM albums WHERE id = ?').get(id);
}

// ─── Core ────────────────────────────────────────────────────────────────────

export function getSongs({ search, artist, album, year, status, tags, printReady, sort = 'title', page = 1, limit = 50 } = {}) {
  let sql = 
    SELECT s., ar.name AS artistname, al.title AS albumtitle
    FROM songs s
    LEFT JOIN artists ar ON s.artistid = ar.id
    LEFT JOIN albums  al ON s.albumid  = al.id
    WHERE 1=1
  ;
  const params = [];

  if (search) {
    sql += ' AND (s.normalizedtitle LIKE ? OR ar.normalized LIKE ?)';
    const q = %${normalize(search)}%;
    params.push(q, q);
  }
  if (artist)     { sql += ' AND ar.name = ?';            params.push(artist); }
  if (album)      { sql += ' AND al.title = ?';           params.push(album); }
  if (year)       { sql += ' AND s.year = ?';             params.push(year); }
  if (status)     { sql += ' AND s.lyricsstatus = ?';    params.push(status); }
  if (printReady !== undefined) {
    sql += ' AND s.isprintready = ?';
    params.push(printReady ? 1 : 0);
  }

  if (tags?.length) {
    sql += 
      AND s.id IN (
        SELECT songid FROM songtags st
        JOIN tags t ON st.tagid = t.id
        WHERE t.name IN (${tags.map(() => '?').join(',')})
        GROUP BY songid HAVING COUNT() = ${tags.length}
      )
    ;
    params.push(...tags);
  }

  const allowed = { title: 's.normalizedtitle', artist: 'ar.normalized', year: 's.year' };
  sql +=  ORDER BY ${allowed[sort] || 's.normalizedtitle'};
  sql +=  LIMIT ? OFFSET ?;
  params.push(limit, (page - 1)  limit);

  return db.prepare(sql).all(...params);
}

export function getSongById(id) {
  const song = db.prepare(
    SELECT s., ar.name AS artistname, al.title AS albumtitle
     FROM songs s
     LEFT JOIN artists ar ON s.artistid = ar.id
     LEFT JOIN albums  al ON s.albumid  = al.id
     WHERE s.id = ?
  ).get(id);

  if (!song) return null;

  song.lyrics = db.prepare(
    'SELECT  FROM lyrics WHERE songid = ? ORDER BY updatedat DESC LIMIT 1'
  ).get(id) || null;

  song.tags = db.prepare(
    SELECT t.name FROM tags t
     JOIN songtags st ON t.id = st.tagid
     WHERE st.songid = ?
  ).all(id).map(r => r.name);

  return song;
}

export function createSong(data) {
  const artist = upsertArtist(data.artist || 'Unknown');
  const album  = data.album
    ? upsertAlbum({ title: data.album, artistId: artist.id, year: data.year, coverUrl: data.coverurl, spotifyId: data.albumspotifyid })
    : null;

  const id = uuidv4();
  db.prepare(
    INSERT INTO songs
      (id, title, normalizedtitle, artistid, albumid, tracknumber,
       year, language, spotifyid, spotifyurl, coverurl, lyricsstatus)
     VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 'missing')
  ).run(
    id,
    sanitizeText(data.title),
    normalize(data.title),
    artist.id,
    album?.id || null,
    data.tracknumber || null,
    data.year || null,
    data.language || 'unknown',
    data.spotifyid || null,
    data.spotifyurl || null,
    data.coverurl || null,
  );

  return getSongById(id);
}

export function updateSong(id, data) {
  const fields = [];
  const values = [];

  const allowed = ['title', 'year', 'language', 'versiontype', 'isprintready'];
  for (const key of allowed) {
    if (data[key] !== undefined) {
      fields.push(${key} = ?);
      values.push(typeof data[key] === 'string' ? sanitizeText(data[key]) : data[key]);
    }
  }

  if (data.title) {
    fields.push('normalizedtitle = ?');
    values.push(normalize(data.title));
  }

  if (data.lyricsstatus) {
    fields.push('lyricsstatus = ?');
    values.push(data.lyricsstatus);
  }

  fields.push("updatedat = datetime('now')");
  values.push(id);

  db.prepare(UPDATE songs SET ${fields.join(', ')} WHERE id = ?).run(...values);
  return getSongById(id);
}

export function deleteSong(id) {
  db.prepare('DELETE FROM songs WHERE id = ?').run(id);
}

// ─── Lyrics ──────────────────────────────────────────────────────────────────

export function saveLyrics(songId, { text, source = 'manual', confidenceScore = 1, isVerified = 0 }) {
  const existing = db.prepare('SELECT id FROM lyrics WHERE songid = ?').get(songId);
  const cleanText = sanitizeText(text);
  const status = isVerified ? 'reviewed'
    : source === 'manual' ? 'manual'
    : confidenceScore >= CONFIDENCEREVIEWTHRESHOLD ? 'auto'
    : 'missing';

  if (existing) {
    db.prepare(
      UPDATE lyrics SET text = ?, source = ?, confidencescore = ?,
       isverified = ?, updatedat = datetime('now')
       WHERE songid = ?
    ).run(cleanText, source, confidenceScore, isVerified ? 1 : 0, songId);
  } else {
    db.prepare(
      INSERT INTO lyrics (id, songid, text, source, confidencescore, isverified)
       VALUES (?, ?, ?, ?, ?, ?)
    ).run(uuidv4(), songId, cleanText, source, confidenceScore, isVerified ? 1 : 0);
  }

  db.prepare(UPDATE songs SET lyricsstatus = ?, updatedat = datetime('now') WHERE id = ?)
    .run(status, songId);

  return getSongById(songId);
}

export async function autoFetchLyrics(songId) {
  const song = getSongById(songId);
  if (!song) throw new Error('Song not found');

  const result = await fetchLyrics(song.title, song.artistname);
  if (!result) return { fetched: false, song };

  const updated = saveLyrics(songId, {
    text:            result.lyricstext,
    source:          result.source,
    confidenceScore: result.confidencescore,
    isVerified:      0,
  });

  return { fetched: true, song: updated };
}

// ─── Tags ────────────────────────────────────────────────────────────────────

export function setTags(songId, tagNames) {
  const clean = tagNames.map(t => sanitizeText(t.trim().toLowerCase())).filter(Boolean);

  db.prepare('DELETE FROM songtags WHERE songid = ?').run(songId);

  for (const name of clean) {
    let tag = db.prepare('SELECT id FROM tags WHERE name = ?').get(name);
    if (!tag) {
      const id = uuidv4();
      db.prepare('INSERT INTO tags (id, name) VALUES (?, ?)').run(id, name);
      tag = { id };
    }
    db.prepare('INSERT OR IGNORE INTO songtags (songid, tagid) VALUES (?, ?)').run(songId, tag.id);
  }
}

// ─── Bulk ────────────────────────────────────────────────────────────────────

export function bulkUpdate(ids, data) {
  const updateMany = db.transaction(() => {
    for (const id of ids) updateSong(id, data);
  });
  updateMany();
}

// ─── Duplicates ──────────────────────────────────────────────────────────────

export function findDuplicates() {
  const songs = db.prepare(
    SELECT s.id, s.normalizedtitle, ar.normalized AS artistnorm
     FROM songs s LEFT JOIN artists ar ON s.artistid = ar.id
  ).all();

  const groups = [];
  const used   = new Set();

  for (let i = 0; i < songs.length; i++) {
    if (used.has(songs[i].id)) continue;
    const group = [songs[i]];
    for (let j = i + 1; j < songs.length; j++) {
      if (
        songs[i].normalizedtitle === songs[j].normalizedtitle &&
        songs[i].artistnorm      === songs[j].artistnorm
      ) {
        group.push(songs[j]);
        used.add(songs[j].id);
      }
    }
    if (group.length > 1) {
      used.add(songs[i].id);
      groups.push(group.map(s => getSongById(s.id)));
    }
  }

  return groups;
}

export function mergeSongs(keepId, mergeIds, { useMetadataFrom, useLyricsFrom }) {
  const merge = db.transaction(() => {
    const meta   = getSongById(useMetadataFrom);

      year:        meta.year,

import db             from '../db/index.js';
import { createSong } from './songService.js';
import { normalize }  from '../utils/normalize.js';

export function importSpotifyTracks(tracks) {
  const results = { imported: 0, skipped: 0, errors: [] };

  const importAll = db.transaction(() => {
    for (const track of tracks) {
      try {
        // Skip if spotifyid already exists
        const exists = db.prepare(
          'SELECT id FROM songs WHERE spotifyid = ?'
        ).get(track.spotifyid);

        if (exists) { results.skipped++; continue; }

        // Check for title+artist near-duplicate
        const normTitle  = normalize(track.title);
        const normArtist = normalize(track.artist);
        const nearDupe = db.prepare(
          SELECT s.id FROM songs s
           JOIN artists ar ON s.artistid = ar.id
           WHERE s.normalizedtitle = ? AND ar.normalized = ?
        ).get(normTitle, normArtist);

        if (nearDupe) { results.skipped++; continue; }

        createSong(track);
        results.imported++;
      } catch (err) {
        results.errors.push({ track: track.title, error: err.message });
      }
    }
  });

  importAll();
  return results;
}

export function importFromJSON(records) {
  const results = { imported: 0, skipped: 0, errors: [] };
  const importAll = db.transaction(() => {
    for (const record of records) {
      try {
        if (!record.title) { results.errors.push({ record, error: 'Missing title' }); continue; }
        createSong(record);
        results.imported++;
      } catch (err) {
        results.errors.push({ record: record.title, error: err.message });
      }
    }
  });
  importAll();
  return results;
}

import { v4 as uuidv4 } from 'uuid';
import db from '../db/index.js';
import { sanitizeText } from '../utils/sanitize.js';

export function getCollections() {
  return db.prepare('SELECT  FROM collections ORDER BY name').all();
}

export function getCollection(id) {
  const col = db.prepare('SELECT  FROM collections WHERE id = ?').get(id);
  if (!col) return null;
  col.songs = db.prepare(
    SELECT s., ar.name AS artistname
     FROM songs s
     JOIN collectionsongs cs ON s.id = cs.songid
     JOIN artists ar ON s.artistid = ar.id
     WHERE cs.collectionid = ?
     ORDER BY cs.position
  ).all(id);
  return col;
}

export function createCollection(name, description = '') {
  const id = uuidv4();
  db.prepare(
    'INSERT INTO collections (id, name, description) VALUES (?, ?, ?)'
  ).run(id, sanitizeText(name), sanitizeText(description));
  return getCollection(id);
}

export function addSongToCollection(collectionId, songId) {
  const maxPos = db.prepare(
    'SELECT MAX(position) AS m FROM collectionsongs WHERE collectionid = ?'
  ).get(collectionId)?.m ?? -1;
  db.prepare(
    'INSERT OR IGNORE INTO collectionsongs (collectionid, songid, position) VALUES (?, ?, ?)'
  ).run(collectionId, songId, maxPos + 1);
}

export function removeSongFromCollection(collectionId, songId) {
  db.prepare(
    'DELETE FROM collectionsongs WHERE collectionid = ? AND songid = ?'
  ).run(collectionId, songId);
}

backend/src/print/engine.js

`js
import puppeteer from 'puppeteer';
import { sanitizeText } from '../utils/sanitize.js';

/
  config shape:
  {
    format: 'A4' | 'A5',
    columns: 1 | 2,
    fontSize: 'small' | 'medium' | 'large',
    lineSpacing: 'tight' | 'normal' | 'loose',
    margins: 'narrow' | 'normal' | 'wide',
    cleanMode: boolean,
    performanceMode: boolean,
    includeToc: boolean,
    includeIndex: boolean,
    breakMode: 'one-per-page' | 'continuous',
  }
 /

const FONTSIZES = { small: '11px', medium: '13px', large: '16px' };
const LINEHEIGHTS = { tight: 1.4, normal: 1.7, loose: 2.1 };
const MARGINSIZES = { narrow: '12mm', normal: '18mm', wide: '25mm' };

function formatLyrics(text) {
  // Escape HTML, then restore newlines as <br>
  return sanitizeText(text)
    .replace(/\n{2,}/g, '</p><p class="verse">')
    .replace(/\n/g, '<br>');
}

function songHtml(song, config) {
  const lyrics   = song.lyrics?.text || '';
  const bodyHtml = lyrics
    ? <div class="lyrics"><p class="verse">${formatLyrics(lyrics)}</p></div>
    : <p class="missing">— lyrics not available —</p>;

  const header = config.cleanMode ? '' : 
    <div class="song-header">
      <h2 class="song-title">${sanitizeText(song.title)}</h2>
      <p  class="song-meta">${sanitizeText(song.artistname)}${
        song.albumtitle ?  · <em>${sanitizeText(song.albumtitle)}</em> : ''
      }</p>
    </div>;

  const breakClass = config.breakMode === 'one-per-page' ? 'page-break' : 'song-block';
  return <div class="${breakClass}">${header}${bodyHtml}</div>;
}

function tocHtml(songs) {
  const rows = songs
    .map((s, i) => <tr><td>${i + 1}</td><td>${sanitizeText(s.title)}</td><td>${sanitizeText(s.artistname)}</td></tr>)
    .join('');
  return 
    <div class="page-break toc">
      <h2>Table of Contents</h2>
      <table><thead><tr><th>#</th><th>Title</th><th>Artist</th></tr></thead>
      <tbody>${rows}</tbody></table>
    </div>;
}

function indexHtml(songs) {
  const sorted = [...songs].sort((a, b) => a.title.localeCompare(b.title));
  const rows = sorted
    .map(s => <tr><td>${sanitizeText(s.title)}</td><td>${sanitizeText(s.artistname)}</td></tr>)
    .join('');
  return 
    <div class="page-break index">
      <h2>Alphabetical Index</h2>
      <table><thead><tr><th>Title</th><th>Artist</th></tr></thead>
      <tbody>${rows}</tbody></table>
    </div>;
}

function buildHtml(songs, config) {
  const fontSize   = FONTSIZES[config.fontSize]   || FONTSIZES.medium;
  const lineHeight = LINEHEIGHTS[config.lineSpacing] || LINEHEIGHTS.normal;
  const margin     = MARGINSIZES[config.margins]   || MARGINSIZES.normal;
  const columns    = config.performanceMode ? 1 : (config.columns || 1);
  const perfSize   = config.performanceMode ? '18px' : fontSize;
  const perfLH     = config.performanceMode ? 2.2 : lineHeight;

  const songsHtml  = songs.map(s => songHtml(s, config)).join('\n');
  const frontMatter = [
    config.includeToc   ? tocHtml(songs)   : '',
    config.includeIndex ? indexHtml(songs) : '',
  ].join('');

  return <!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<style>
  @page {
    size: ${config.format || 'A4'};
    margin: ${margin};
  }
   { box-sizing: border-box; }
  body {
    font-family: Georgia, serif;
    font-size: ${perfSize};
    line-height: ${perfLH};
    color: #111;
    columns: ${columns};
    column-gap: 12mm;
  }
  .page-break {
    break-before: page;
    break-inside: avoid;
    column-span: all;
  }
  .song-block {
    break-inside: avoid;
    margin-bottom: 8mm;
  }
  .song-header {
    break-after: avoid;
    margin-bottom: 3mm;
  }
  .song-title {
    font-size: 1.3em;
    font-weight: bold;
    margin: 0 0 1mm 0;
  }
  .song-meta {
    font-size: 0.85em;
    color: #555;
    margin: 0 0 2mm 0;
  }
  .lyrics { break-inside: avoid-page; }
  .verse {
    margin: 0 0 3mm 0;
    break-inside: avoid;
  }
  .missing { color: #999; font-style: italic; }
  table { width: 100%; border-collapse: collapse; font-size: 0.9em; }
  th, td { border-bottom: 1px solid #ccc; padding: 2mm 3mm; text-align: left; }
  th { background: #f5f5f5; }
  .toc h2, .index h2 { margin-bottom: 4mm; }
</style>
</head>
<body>
${frontMatter}
${songsHtml}
</body>
</html>;
}

export async function generatePdf(songs, config = {}) {
  const html = buildHtml(songs, config);

  const browser = await puppeteer.launch({
    headless: 'new',
    args: ['--no-sandbox', '--disable-setuid-sandbox'],
  });

  try {
    const page = await browser.newPage();
    await page.setContent(html, { waitUntil: 'networkidle0' });

    const pdf = await page.pdf({
      format:           config.format || 'A4',
      printBackground:  true,
      displayHeaderFooter: !config.cleanMode,
      headerTemplate:  '<span></span>',
      footerTemplate: 
        <div style="font-size:9px;width:100%;text-align:center;color:#999;margin:0 18mm;">
          <span class="pageNumber"></span> / <span class="totalPages"></span>
        </div>,
      margin: { top: '10mm', bottom: '14mm' },
    });

    return pdf;
  } finally {
    await browser.close();
  }
}

const BASE = import.meta.env.VITEAPIURL || 'http://localhost:3001/api';

async function request(path, options = {}) {
  const res = await fetch(${BASE}${path}, {
    headers: { 'Content-Type': 'application/json', ...options.headers },
    ...options,
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: res.statusText }));
    throw new Error(err.error || HTTP ${res.status});
  }
  if (res.status === 204) return null;
  return res.json();
}

export const api = {
  // Songs
  getSongs:      (params = {}) => request('/songs?' + new URLSearchParams(
    Object.fromEntries(Object.entries(params).filter(([,v]) => v !== undefined && v !== ''))
  )),
  getSong:       (id)          => request(/songs/${id}),
  createSong:    (data)        => request('/songs',    { method: 'POST', body: JSON.stringify(data) }),
  updateSong:    (id, data)    => request(/songs/${id}, { method: 'PATCH', body: JSON.stringify(data) }),
  deleteSong:    (id)          => request(/songs/${id}, { method: 'DELETE' }),
  saveLyrics:    (id, data)    => request(/songs/${id}/lyrics, { method: 'PUT', body: JSON.stringify(data) }),
  fetchLyrics:   (id)          => request(/songs/${id}/fetch-lyrics, { method: 'POST' }),
  setTags:       (id, tags)    => request(/songs/${id}/tags, { method: 'PUT', body: JSON.stringify({ tags }) }),
  bulkUpdate:    (ids, data)   => request('/songs/bulk', { method: 'POST', body: JSON.stringify({ ids, data }) }),
  getDuplicates: ()            => request('/songs/duplicates'),
  mergeSongs:    (data)        => request('/songs/merge', { method: 'POST', body: JSON.stringify(data) }),

  // Spotify
  importPlaylist: (playlistId) => request('/spotify/import/playlist', { method: 'POST', body: JSON.stringify({ playlistId }) }),
  importAlbum:    (albumId)    => request('/spotify/import/album',    { method: 'POST', body: JSON.stringify({ albumId }) }),

  // Collections
  getCollections:    ()             => request('/collections'),
  getCollection:     (id)           => request(/collections/${id}),
  createCollection:  (name, desc)   => request('/collections', { method: 'POST', body: JSON.stringify({ name, description: desc }) }),
  addToCollection:   (colId, songId) => request(/collections/${colId}/songs, { method: 'POST', body: JSON.stringify({ songId }) }),
  removeFromCollection: (colId, songId) => request(/collections/${colId}/songs/${songId}, { method: 'DELETE' }),

  // Print
  printPdf: async (body) => {
    const res = await fetch(${BASE}/print/pdf, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error('PDF generation failed');
    const blob = await res.blob();
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url; a.download = 'songbook.pdf'; a.click();
    URL.revokeObjectURL(url);
  },
};

    }
    setSelected(new Set()); load();

    api.printPdf({

          <button style={styles.smBtn} onClick={bulkFetchLyrics}>Fetch Lyrics</button>

import { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { api } from '../api/client.js';

const STATUSLABELS = { missing: '❌ Missing', auto: '🤖 Auto', manual: '✏️ Manual', reviewed: '✅ Reviewed' };

export default function SongPage() {
  const { id }   = useParams();
  const navigate = useNavigate();
  const isNew    = id === 'new';

  const [song,   setSong]   = useState(null);
  const [form,   setForm]   = useState({ title: '', artist: '', album: '', year: '', language: '' });
  const [lyrics, setLyrics] = useState('');
  const [saving, setSaving] = useState(false);
  const [msg,    setMsg]    = useState('');

  useEffect(() => {
    if (isNew) return;
    api.getSong(id).then(s => {
      setSong(s);
      setForm({ title: s.title, artist: s.artistname, album: s.albumtitle || '', year: s.year || '', language: s.language || '' });
      setLyrics(s.lyrics?.text || '');
    });
  }, [id, isNew]);

  const save = async () => {
    setSaving(true); setMsg('');
    try {
      if (isNew) {
        const created = await api.createSong(form);
        if (lyrics) await api.saveLyrics(created.id, { text: lyrics });
        navigate(/songs/${created.id});
      } else {
        await api.updateSong(id, form);
        if (lyrics !== (song?.lyrics?.text || ''))
          await api.saveLyrics(id, { text: lyrics });
        const updated = await api.getSong(id);
        setSong(updated);
        setMsg('Saved.');
      }
    } catch (e) { setMsg(Error: ${e.message}); }
    finally { setSaving(false); }
  };

  const fetchLyrics = async () => {
    setSaving(true); setMsg('Fetching…');
    try {
      const result = await api.fetchLyrics(id);
      if (result.fetched) {
        setLyrics(result.song.lyrics?.text || '');
        setSong(result.song);
        setMsg('Lyrics fetched.');
      } else {
        setMsg('No lyrics found by any provider.');
      }
    } catch (e) { setMsg(Error: ${e.message}); }
    finally { setSaving(false); }
  };

  const togglePrintReady = async () => {
    const updated = await api.updateSong(id, { isprintready: song.isprintready ? 0 : 1 });
    setSong(updated);
  };

  const markVerified = async () => {
    await api.saveLyrics(id, { text: lyrics, isverified: 1 });
    const updated = await api.getSong(id);
    setSong(updated); setMsg('Marked as reviewed.');
  };

  return (
    <div style={styles.page}>
      <button style={styles.back} onClick={() => navigate('/')}>← Library</button>

      <h2 style={{ marginBottom: 4 }}>{isNew ? 'New Song' : (song?.title || '…')}</h2>

      {!isNew && song && (
        <div style={styles.meta}>
          <span>Status: <strong>{STATUSLABELS[song.lyricsstatus]}</strong></span>
          <span>Print ready: <strong>{song.isprintready ? 'Yes' : 'No'}</strong></span>
        </div>
      )}

      <div style={styles.grid}>
        {[
          ['Title',    'title',    'text'],
          ['Artist',   'artist',   'text'],
          ['Album',    'album',    'text'],
          ['Year',     'year',     'number'],
          ['Language', 'language', 'text'],
        ].map(([label, key, type]) => (
          <label key={key} style={styles.label}>
            {label}
            <input style={styles.input} type={type} value={form[key] || ''}
              onChange={e => setForm(f => ({ ...f, [key]: e.target.value }))} />
          </label>
        ))}
      </div>

      <label style={{ ...styles.label, display: 'block', marginTop: 16 }}>
        Lyrics
        <textarea style={styles.textarea} value={lyrics}
          onChange={e => setLyrics(e.target.value)}
          placeholder="Paste or type lyrics here…" />
      </label>

      <div style={styles.btnRow}>
        <button style={styles.primary} onClick={save} disabled={saving}>
          {saving ? 'Saving…' : 'Save'}
        </button>
        {!isNew && (
          <>
            <button style={styles.secondary} onClick={fetchLyrics} disabled={saving}>
              Auto-fetch Lyrics
            </button>
            <button style={styles.secondary} onClick={markVerified}>
              Mark Reviewed
            </button>
            <button style={styles.secondary} onClick={togglePrintReady}>
              {song?.isprintready ? 'Unmark Print Ready' : 'Mark Print Ready'}
            </button>

              Print This Song
            </button>
          </>
        )}
      </div>

      {msg && <p style={{ color: msg.startsWith('Error') ? 'red' : '#27ae60', marginTop: 8 }}>{msg}</p>}
    </div>
  );

Key Design Decisions

| Decision | Rationale |
|---|---|
| SQLite + better-sqlite3 | Synchronous, zero-config, fast for personal scale |
| Provider pattern for lyrics | Swap or add sources without touching business logic |
| Puppeteer for PDF | Full CSS layout control; handles pagination, columns, fonts correctly |
| Confidence scoring | Auto-fetched lyrics below threshold stay in "needs review" rather than silently polluting data |
| Normalize + fuzzy match | Handles "feat." variants, punctuation differences without a heavy NLP dep |
| Sanitize at write time | XSS protection happens once at input, not at every render |

The most interesting extension point here is the lyrics provider interface — adding a new source (say, Musixmatch via its official API) is just a new class that extends BaseLyricsProvider` and gets dropped into the providers array, with zero changes elsewhere.
